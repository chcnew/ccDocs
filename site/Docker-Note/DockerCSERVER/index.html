    二、docker使用

    # 镜像列表
docker images
# 拉取镜像
docker pull www.xxx.com:880/ttt-xprs/cserver-env/ubuntu:latest
# 删除镜像
docker rmi <IMAGE-ID>

# 创建新的容器
docker run -itd www.xxx.com:880/ttt-xprs/cserver-env/ubuntu:latest /bin/bash

# 查看全部容器列表 
docker ps -a

# 启动容器
docker start <CONTAINER-ID>

# 停止容器
docker stop <CONTAINER-ID>

# 容器日志
docker logs <CONTAINER-ID>

# 删除容器
docker rm <CONTAINER-ID or IMAGE>

# 进入容器
docker exec -it <CONTAINER-ID> /bin/bash

# 保留容器为镜像
docker commit <container_id> cserver-env:latest

    Dockerfile：

    # 基础镜像
FROM www.xxx.com:880/ttt-xprs/cserver-env/ubuntu:base1.1

# 维护者信息
MAINTAINER xxx ttt@ccc.com

# 暴露端口
EXPOSE 80 8810-8819

# 准备工作
RUN mkdir -p /data/Local && chmod 755 -R /data/ && \
    mkdir -p /root/DAY/ && \
    mkdir -p /root/DAW/ && \
    mkdir -p /hostip/ && \
    python3 -m pip install --upgrade pip  && \
    python3 -m pip uninstall -y xxx && \
    pip3 install --user xxx

# 复制文件
COPY realip /hostip/
COPY run_env.sh /root/

# 权限设置
RUN chmod +x /root/run_env.sh

# 工作目录
WORKDIR /root/

# 启动进程
ENTRYPOINT ["/root/run_env.sh"]


    # 当前目录中查找 Dockerfile 文件，并根据 Dockerfile 中的指令构建出一个镜像
docker build -t cserver-env:latest .

    这句命令是在 Docker 环境中启动一个新的容器，并进入该容器的交互式终端。具体解释如下： - docker run 是启动容器的命令。 - -it 参数表示在交互模式下运行容器，并分配一个伪终端。这使得用户可以在容器中执行命令，并与容器进行交互。 - --rm 参数表示容器退出后自动删除容器。这可以防止容器在退出后占用过多的磁盘空间。 - cserver-env:latest 是要启动的 Docker 镜像的名称和版本号。这里的 cserver-env 是镜像的名称，:latest 表示使用最新版本的镜像。 - /bin/bash 是要在容器中运行的命令。这里的 /bin/bash 表示在容器中启动 Bash Shell

    docker run -it --rm cserver-env:latest /bin/bash

    三、磁盘挂载

    # 要在docker中挂载硬盘，可以使用以下命令(其中，`/host/path`是宿主机上的路径，`/container/path`是容器内的路径，`<image_name>`是要运行的镜像名称。)：
docker run -itd -v /host/path:/container/path <image_name> /bin/bash
# 实际示例
docker run -itd -v /data/Local:/data/Local www.xxx.com:880/ttt-xprs/cserver-env/ubuntu /bin/bash

# 再次查看容器列表 
docker ps -a

    六、免密登录配置

    # 免交互生成密钥
# -q ： 安静模式，一般用在生成密钥上
# -N： 生成一个新密钥
# -f : 生成密钥的文件路径 && # 服务器端加入认证列表
ssh-keygen -q -N " " -f ~/.ssh/id_rsa && cat ~/.ssh/id_rsa.pub > ~/.ssh/authorized_keys

# 启动容器时共享~/.ssh目录
-v ${HOME}/.ssh:/root/.ssh

# 容器ssh至宿主机
chmod 700 /root/.ssh/
chmod 600 /root/.ssh/*
ssh -i /root/.ssh/cida_docker_key -o StrictHostKeyChecking=no root@192.168.127.150 -p 22 "python3 -V"

    systemctl报错

    System has not been booted with systemd as init system (PID 1). Can't operate. Failed to connect to bus: Host is down

    如何处理？

    一、特权参数执行
docker run -itd --privileged=true --name my_centos my-centos /usr/sbin/init
docker exec -it my_centos /bin/bash

二、替换systemctl
wget https://github.com/gdraheim/docker-systemctl-replacement/tree/master/files/docker/systemctl3.py -O /bin/systemctl
chmod a+x /bin/systemctl

    虚拟机VM配置ubuntu22.04

    # 不存在ens33网卡
ifconfig ens33 up
# 配置网络
cd /etc/netplan/
ll
# 配置内容 -------------
# This is the network config written by 'subiquity'
network:
  version: 2
  renderer: NetworkManager  # 渲染器
  ethernets:
    ens33:  # 网卡名称
      dhcp4: no  # 关闭dhcp4
      dhcp6: no  # 关闭dhcp6
      addresses: [192.168.127.150/24]  # 静态ip
      gateway4: 192.168.127.2  # NTA设置网关
      nameservers:
        addresses: [192.168.127.1,114.114.114.114] # DNS设置必须通过共享网络设置的ip
# ---------------------
# 网络配置0、1、2
虚拟机网络配置子网IP：192.168.127.0 掩码：255.255.255.0
共享网络给net8，设置ip：192.168.127.1
NTA设置网关：192.168.127.2
#VM相关任务管理器3个服务需要打开
# netplan服务加载配置
netplan apply
